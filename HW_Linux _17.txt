Написать скрипт, который будет бесконечно пинговать указанный адрес (переменная или ввод пользователя) с интервалом 1 секунда между попытками.
 Если время пинга превышает 100 мс или не удается выполнить пинг в течение 3 последовательных отправок пакетов, скрипт просто выведет сообщения об этом. 
Пришлите скрипт или ссылку на github репо с решением

РЕШЕНИЕ;
#!/bin/bash

read -p "Введите Адрес для пинга: " HOST

fail_count=0

while true

do

    ping_result=$(ping -c 1 -W 1 "$HOST" 2>/dev/null)

    time=$(date "+%Y-%m-%d %H:%M:%S")

        if [[ $ping_result == *"1 received"* ]]

        then

                time_ms=$(echo "$ping_result" | grep "time=" | awk -F'time=' '{print $2}' | awk$

                if [[ $time_ms > 100 ]]; then

                echo "[$time] High latency: $time_ms ms (exceeds 100 ms)"

                else

                echo "[$time] Normal: $time_ms ms"

                fi

        else

                ((fail_count++))

                if [[ $fail_count -eq 3 ]]; then

                echo "Кол-во неудачных попыток равно 3: нет ответа от хоста $HOST"

                fi

        fi

sleep 1

done
